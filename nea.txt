function signup():
	do:
		username = input("Enter a unique username")
		email = input("Enter your unique email")
		print("Password must be at least 8 characters long.")
		print("Password must have 1 uppercase letter and 1 lowercase letter")
		print("Password must have 1 number and 1 special character")
		password = input("Enter a password")
		confirm_password = input("Confirm password")
	until:
        username.is_unique() AND email.is_unique() AND password == confirm_password:
	
	user_database.store( username,email,password.hash() )

	return redirect login

function login():
    do:
        email = input("Enter your unique email")
        password = input("Enter your password")
    until:
        user_database.contains( email, password.hash() )
    
    user.authenticated()
    return redirect home
